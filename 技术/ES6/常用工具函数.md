字节数格式化

```javascript
function formatBytes(bytes, decimals = 2) {
    if (bytes === 0) return '0 Bytes';

    const k = 1024;
    const dm = decimals < 0 ? 0 : decimals;
    const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB', 'PB', 'EB', 'ZB', 'YB'];

    const i = Math.floor(Math.log(bytes) / Math.log(k));

    return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
}

// formatBytes(bytes,decimals)

formatBytes(1024);       // 1 KB
formatBytes('1024');     // 1 KB
formatBytes(1234);       // 1.21 KB
formatBytes(1234, 3);    // 1.205 KB
```

```javascript
function clone(obj) {
    var copy;

    if (null == obj || "object" != typeof obj) return obj;

    if (obj instanceof Date) {
        copy = new Date();
        copy.setTime(obj.getTime());
        return copy;
    }

    if (obj instanceof Array) {
        copy = [];
        for (var i = 0, len = obj.length; i < len; i++) {
            copy[i] = clone(obj[i]);
        }
        return copy;
    }

    if (obj instanceof Object) {
        copy = {};
//         for (var attr in obj) {
//             if (obj.hasOwnProperty(attr)) copy[attr] = clone(obj[attr]);
//         }
        let keys = Reflect.ownKeys(obj);
        keys.forEach(attr=>{
            copy[attr] = clone(obj[attr])
        })
        return copy;
    }

    throw new Error("Unable to copy obj! Its type isn't supported.");
}
```

并发控制

```javascript
const tasks = [
    delay(500),
    delay(400),
    delay(200),
    delay(600)
];

const currency = 2;
let index = 0;
let running = 0;
let count = 0;

function delay(ms) {
    return cb => {
        setTimeout(() => {
            console.log(ms, '---');
            cb();
        }, ms);
    };
}

function run() {
    while(running<currency && index<tasks.length) {
        const task = tasks[index++]
        task(() => {
            if (++count === tasks.length) {
                return finish();
            }
            running--;
            run();
        });
        running++;
    }
}

function finish() {
    console.log('finished');
}

run();
```

promisify

```javascript
function promisify(callbackApi) {
    return function(...args) {
        return new Promise((resolve, reject) => {
            const newArgs = [...args, (err, data) => {
                if (err) {
                    return reject(err);
                }
                resolve(data);
            }];
            callbackApi(...newArgs);
        });
    };
}
```

查找html文本并高亮替换

```javascript
function replaceHtml(html,r,elemObj) {
    let start = 0;
    let newhtml = '';
    const matchs = html.matchAll(r);
    for (let i of matchs) {
        newhtml += html.slice(start, i.index);
        newhtml += createElemString(elemObj, i[0]);
        start = i.index + i[0].length;
    }
    newhtml += html.slice(start);
    return newhtml;
}

function createElemString(elemObj, placer) {
    const { styles = '', className = '', tag = 'span', attrs = {} } = elemObj || {};
    let str = `<${tag}`;
    // 优先应用className
    if (className) {
        str += ` class="${className}"`;
    } else if(styles) {
        str += ` style="${styles};"`
    }
    const attrsStr = Object.keys(attrs).reduce((acc, i) => {
        acc += ` ${i}="${attrs[i]}"`;
        return acc;
    }, '');
    str += attrsStr+'>';
    str += placer;
    str += `</${tag}>`;
    return str;
}

//使用示例
const html = $0.innerHTML;
cosnt r = /w\w+e/g;
const elemObj = {
    styles: 'color:red'，
    attrs: {a:1,b:0,c:'what'}
};
const newhtml = replaceHtml(html, r, elemObj);
$0.innerHTML = newhtml;
```

转换为iso时间

```javascript
function toIsoString(date) {
  var tzo = -date.getTimezoneOffset(),
      dif = tzo >= 0 ? '+' : '-',
      pad = function(num) {
          return (num < 10 ? '0' : '') + num;
      };

  return date.getFullYear() +
      '-' + pad(date.getMonth() + 1) +
      '-' + pad(date.getDate()) +
      'T' + pad(date.getHours()) +
      ':' + pad(date.getMinutes()) +
      ':' + pad(date.getSeconds()) +
      dif + pad(Math.floor(Math.abs(tzo) / 60)) +
      ':' + pad(Math.abs(tzo) % 60);
}

var dt = new Date();
console.log(toIsoString(dt));
```
